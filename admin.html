⚡ One-time setup required — run this SQL in your Supabase SQL Editor to enable news management: 
CREATE TABLE IF NOT EXISTS public.site_news (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  headline TEXT NOT NULL,
  source TEXT NOT NULL,
  date TEXT NOT NULL,
  tag TEXT DEFAULT 'tools',
  url TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE public.site_news ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Anyone can read news" ON public.site_news FOR SELECT USING (true);
CREATE POLICY "Admins can manage news" ON public.site_news FOR ALL USING (
  EXISTS (SELECT 1 FROM public.profiles WHERE user_id = auth.uid() AND is_admin = true)
);
