⚡ One-time setup — run this SQL in your Supabase SQL Editor to enable reflection sync: 
CREATE TABLE IF NOT EXISTS public.reflections (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  lesson_id INT NOT NULL,
  slide_idx INT NOT NULL,
  lesson_title TEXT,
  text TEXT NOT NULL,
  updated_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(user_id, lesson_id, slide_idx)
);
ALTER TABLE public.reflections ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own reflections" ON public.reflections
  FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "Admins read all reflections" ON public.reflections
  FOR SELECT USING (
    EXISTS (SELECT 1 FROM public.profiles WHERE user_id = auth.uid() AND is_admin = true)
  );

⭐ Lesson Ratings — pupil thumbs up/down feedback
▶ First-time setup — run this SQL in Supabase 
CREATE TABLE IF NOT EXISTS public.lesson_ratings (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  lesson_id integer NOT NULL,
  rating text CHECK (rating IN ('up','down')),
  rated_at timestamptz DEFAULT now(),
  UNIQUE(user_id, lesson_id)
);
ALTER TABLE public.lesson_ratings ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users manage own ratings" ON public.lesson_ratings
  FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "Admins read all ratings" ON public.lesson_ratings
  FOR SELECT USING (
    EXISTS (SELECT 1 FROM public.profiles WHERE user_id = auth.uid() AND is_admin = true)
  );

⚡ One-time setup required — run this SQL in your Supabase SQL Editor to enable news management: 
CREATE TABLE IF NOT EXISTS public.site_news (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  headline TEXT NOT NULL,
  source TEXT NOT NULL,
  date TEXT NOT NULL,
  tag TEXT DEFAULT 'tools',
  url TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE public.site_news ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Anyone can read news" ON public.site_news FOR SELECT USING (true);
CREATE POLICY "Admins can manage news" ON public.site_news FOR ALL USING (
  EXISTS (SELECT 1 FROM public.profiles WHERE user_id = auth.uid() AND is_admin = true)
);
